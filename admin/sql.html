SQL Queries

Useful MariaDB queries confirmed against the current redm_vorp schema.

List Characters and Groups

SELECT c.charidentifier, c.firstname, c.lastname, c.`group` AS character_group, u.`group` AS account_group
FROM characters c
LEFT JOIN users u ON u.identifier = c.identifier
ORDER BY c.charidentifier;

Character Money

SELECT charidentifier, firstname, lastname, money
FROM characters
ORDER BY charidentifier;

Character Loadout

SELECT id, name, ammo, used, used2, dropped, label
FROM loadout
WHERE charidentifier = 2
ORDER BY id;

Owned Horses

SELECT id, charidentifier, name, modelname, type, isDefault
FROM stables
ORDER BY charidentifier, id;

Inventory Item

SELECT item_name, amount, inventory_type
FROM character_inventories
WHERE character_id = 2
  AND item_name = 'ammoriflenormal';
